                    i = lastLineI;

                } else {

                    cursor = null;
                    break;
                }

                    cursor = parseEscapedItem(line, items, nextToken.cursor, hasMoreData);
                } else {
                    cursor = parseItem(line, items, i, hasMoreData);
                }
                if (cursor === null) {
                    i = lastLineI;
                    break;
                } else {
                    i = cursor;
                }
            }
 
        }
        cursor !== null && rows.push(items);
        return {line: line.substr(i), rows: rows};
    };
